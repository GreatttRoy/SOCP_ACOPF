\min_{\substack{P_{t,i,j},\, Q_{t,i,j},\, l_{t,i,j},\, v_{t,i} \\ p_{t,i},\, q_{t,i},\, P_{g,t,g},\, Q_{g,t,g},\, P_{DR,t,dr}}} \quad & \sum_{t=0}^{T-1} \sum_{dr \in DR_j} -P_{DR,t,dr} \\
\text{s.t.} \quad \nonumber \\

& q_{t,j} = \sum_{(j,k) \in E} Q_{t,j,k} - \sum_{(i,j) \in E} \left( Q_{t,i,j} - x_{ij} \cdot l_{t,i,j} \right), \quad \forall t,i,j,k \in N \tag{3b} \\
& v_{t,j} = v_{t,i} - 2\left( r_{ij} P_{t,i,j} + x_{ij} Q_{t,i,j} \right) + \left( r_{ij}^2 + x_{ij}^2 \right) l_{t,i,j}, \quad \forall t, (i,j) \in E \tag{4} \\
& \left(2P_{t,i,j}\right)^2 + \left(2Q_{t,i,j}\right)^2 + \left(l_{t,i,j} - v_{t,i}\right)^2 \leq \left(l_{t,i,j} + v_{t,i}\right)^2, \quad \forall t, (i,j) \in E \tag{5} \\

& P_{DR,\min}(t) \leq P_{DR,t,dr} \leq P_{DR,\max}(t), \quad \forall t, dr \in DR \tag{7} \\

\section{基于方向割的锥还原方法}

\subsection{问题背景}

在二阶锥交流潮流优化模型中，二阶锥松弛约束 (5) 对原始非凸约束进行了凸松弛：

\begin{equation}
l_{t,i,j} \cdot v_{t,i} = P_{t,i,j}^2 + Q_{t,i,j}^2 \quad \Rightarrow \quad \left\| \begin{bmatrix} 2P_{t,i,j} \\ 2Q_{t,i,j} \\ l_{t,i,j} - v_{t,i} \end{bmatrix} \right\|_2 \leq l_{t,i,j} + v_{t,i}
\end{equation}

\textbf{松弛的几何意义}：
\begin{itemize}
\item 原始约束要求点 $(P_{t,i,j}, Q_{t,i,j}, l_{t,i,j}, v_{t,i})$ 必须位于\textbf{二阶锥的表面}（锥面约束）
\item 松弛后的约束允许点位于\textbf{二阶锥的内部或表面}（锥约束）
\end{itemize}

在辐射状配电网络中，该松弛通常是紧的（tight），即最优解自然满足等式约束。但在某些情况下，松弛解可能落在锥的内部，此时需要进行\textbf{锥还原}，将解投影到锥面上以获得原问题的可行解。

\subsection{锥松弛间隙检测}

设 SOCP 求解得到的最优解为 $(P^*_{t,i,j}, Q^*_{t,i,j}, l^*_{t,i,j}, v^*_{t,i})$，定义\textbf{锥松弛间隙}为：

\begin{equation}
\Delta_{t,i,j} = l^*_{t,i,j} \cdot v^*_{t,i} - \left[(P^*_{t,i,j})^2 + (Q^*_{t,i,j})^2\right]
\end{equation}

\textbf{间隙判定}：
\begin{itemize}
\item 若 $\Delta_{t,i,j} \approx 0$（在数值误差范围内），则松弛是紧的，无需还原
\item 若 $\Delta_{t,i,j} > \epsilon$（给定容差），则需要进行锥还原
\end{itemize}

\subsection{基于方向的锥还原方法}

当检测到松弛间隙时，采用基于方向的迭代方法将解还原到锥面上。该方法基于 \textbf{Cauchy-Schwarz 不等式},导出\textbf{方向割约束}。

\subsubsection{Cauchy-Schwarz 不等式}

对于支路 $(i,j)$，记 $\mathbf{x}_{ij} = (2P_{t,i,j}, 2Q_{t,i,j}, l_{t,i,j} - v_{t,i})^T$，则二阶锥约束为：

\begin{equation}
\|\mathbf{x}_{ij}\|_2 \leq l_{t,i,j} + v_{t,i}
\end{equation}

对于任意方向向量 $\mathbf{d} = (d_P, d_Q, d_l)^T$，由 Cauchy-Schwarz 不等式可得：

\begin{equation}
\frac{\mathbf{d}^T \mathbf{x}_{ij}}{\|\mathbf{d}\|_2} \leq \|\mathbf{x}_{ij}\|_2 \leq {l_{t,i,j} + v_{t,i}}
\end{equation}

这表明：沿任意方向 $\mathbf{d}$ 的投影不超过 $l_{t,i,j} + v_{t,i}$。

\subsubsection{方向割约束}

为了收紧可行域，在方向 $\mathbf{d}$ 上添加\textbf{带收缩因子的方向割}：

\begin{equation}
\frac{\mathbf{d}^T \mathbf{x}_{ij}}{\|\mathbf{d}\|_2} \geq r \cdot (l_{t,i,j} + v_{t,i})
\end{equation}

其中 $r \in [0, 1]$ 为收缩因子：
\begin{itemize}
\item $r = 1$：最紧的割，仅保留方向 $\mathbf{d}$ 上的锥面点
\item $r < 1$：较松的割，保留方向 $\mathbf{d}$ 附近一定角度范围内的锥面点
\end{itemize}

\subsection{算法总体流程}

层次化方向割锥还原算法由三个阶段组成：初始化、主迭代循环和终止处理。算法以 SOCP 松弛解作为起点，通过逐步添加方向割约束收紧可行域，驱动松弛解向锥面靠近，直至锥松弛间隙低于收敛容差 $\epsilon$ 或达到最大迭代次数 $K_{\max}$。

\paragraph{阶段一：初始化}
以初始 SOCP 解 $(P^*, Q^*, l^*, v^*)$ 为出发点，对每条支路 $(i,j)$ 提取当前锥表示向量方向 $\mathbf{d} = (2P^*_{t,i,j}, 2Q^*_{t,i,j}, l^*_{t,i,j}-v^*_{t,i})^T$，计算初始收缩因子 $r_{\min} = \|\mathbf{d}\| / (l^*_{t,i,j} + v^*_{t,i})$（即当前解在锥面方向上的投影比），并将对应的方向割约束加入约束集合 $\mathcal{L}$，同时置层计数 $\ell = 1$、迭代计数 $k = 0$、间隙历史集合 $\mathcal{H} = \emptyset$。

\paragraph{阶段二：主迭代循环}
每轮迭代包含以下步骤：

\textbf{步骤 1（SOCP 求解）}：在当前割约束集合 $\mathcal{L}$ 下求解 SOCP，得到新解 $(P^k, Q^k, l^k, v^k)$。

\textbf{步骤 2（间隙计算）}：对所有支路和时段计算锥松弛间隙 $\Delta^k_{t,i,j} = l^k_{t,i,j} v^k_{t,i} - [(P^k_{t,i,j})^2 + (Q^k_{t,i,j})^2]$，并取最大间隙 $\Delta^k_{\max}$ 记录至历史集合 $\mathcal{H}$。

\textbf{步骤 3（全局收敛检验）}：若 $\Delta^k_{\max} \leq \epsilon$，则所有支路的松弛均已收紧，返回当前解并标记为收敛。

\textbf{步骤 4（层内自适应方向更新）}：对仍有间隙（$\Delta^k_{t,i,j} > \epsilon$）的支路，以当前解方向 $\hat{\mathbf{d}}$ 和当前层收缩因子 $r_\ell$ 生成新的方向割，补充进 $\mathcal{L}$。收缩因子 $r_\ell$ 在各层间线性递增（从 $r_{\min}$ 到 $1$），层号越高割越紧。

\textbf{步骤 5--6（停滞检测与层次提升）}：若近 $W_{\text{stag}}$ 次迭代的最大间隙改善量均低于 $\epsilon_{\text{stag}}$，判定当前层已停滞。此时若 $\ell < L_{\max}$，则提升至下一层：增大收缩因子 $r_\ell$，生成新层全局方向集 $\mathcal{D}_\ell$ 并添加对应的割约束，清空间隙历史重新监测；否则退出循环。

\paragraph{阶段三：终止处理}
若总迭代次数达到 $K_{\max}$ 仍未收敛，或层数达到最大层数 $L_{\max}$仍未收敛，返回当前最优解并标记为未完全收敛，供后续处理使用。
\subsection{算法伪代码}

\begin{breakablealgorithm}
\small
\caption{层次化方向割锥还原算法}
\label{alg:hierarchical_cone_recovery}
\begin{algorithmic}[1]
\REQUIRE SOCP 松弛解 $(P^*, Q^*, l^*, v^*)$，参数 $\epsilon, \epsilon_{\text{stag}}, W_{\text{stag}}, L_{\max}, K_{\max}$
\ENSURE 还原解 $(\bar{P}, \bar{Q}, \bar{l}, \bar{v})$，收敛状态
\STATE \textit{// 阶段一：初始化}
\STATE $k \leftarrow 0$，$\ell \leftarrow 1$
\STATE $\mathcal{L} \leftarrow \emptyset$ \COMMENT{方向割约束集合}
\STATE $\mathcal{H} \leftarrow \emptyset$ \COMMENT{间隙历史记录}
\STATE $\mathbf{d} \leftarrow (2P^*_{t,i,j}, 2Q^*_{t,i,j}, l^*_{t,i,j}-v^*_{t,i})^T,\ \forall (i,j) \in E,\, t \in [0,T-1]$ \COMMENT{初始解的方向集合}
\STATE $\mathcal{L} \leftarrow \mathcal{L} \cup \left\{\mathbf{d}^T\mathbf{x}_{ij} / \|\mathbf{d}\| \geq r_1 (l_{t,i,j} + v_{t,i})\right\}$ \COMMENT{初始层的方向割约束}
\STATE $r_{\min} \leftarrow \|\mathbf{d}\| / (l^*_{t,i,j}+v^*_{t,i})$ \COMMENT{初始收缩因子}
\STATE \textit{// 阶段二：主迭代循环}
\WHILE{$k < K_{\max}$}
    \STATE $k \leftarrow k + 1$
    \STATE \textit{// 步骤 1：求解带约束的 SOCP 问题}
    \STATE $(P^k, Q^k, l^k, v^k) \leftarrow \text{SOLVE\_SOCP}(\mathcal{L})$
    \STATE \textit{// 步骤 2：计算锥松弛间隙}
    \FORALL{$(i,j) \in E,\ t \in [0,T-1]$}
        \STATE $\Delta^k_{t,i,j} \leftarrow l^k_{t,i,j} \cdot v^k_{t,i} - \left[(P^k_{t,i,j})^2 + (Q^k_{t,i,j})^2\right]$
    \ENDFOR
    \STATE $\Delta^k_{\max} \leftarrow \max_{t,i,j} \Delta^k_{t,i,j}$；$\ \mathcal{H} \leftarrow \mathcal{H} \cup \{\Delta^k_{\max}\}$
    \STATE \textit{// 步骤 3：检查全局收敛}
    \IF{$\Delta^k_{\max} \leq \epsilon$}
        \RETURN $(P^k, Q^k, l^k, v^k)$，\textsc{Converged}
    \ENDIF
    \STATE \textit{// 步骤 4：层内自适应方向搜索（固定 $r_\ell$）}
    \STATE $r_\ell \leftarrow r_{\min} + (\ell-1)/(L_{\max}-1) \cdot (1 - r_{\min})$
    \FORALL{$(i,j) \in E,\ t \in [0,T-1]$ 且 $\Delta^k_{t,i,j} > \epsilon$}
        \STATE $\hat{\mathbf{d}} \leftarrow (2P^k_{t,i,j}, 2Q^k_{t,i,j}, l^k_{t,i,j}-v^k_{t,i})^T / \left\|(2P^k_{t,i,j}, 2Q^k_{t,i,j}, l^k_{t,i,j}-v^k_{t,i})\right\|$
        \STATE $\mathcal{L} \leftarrow \mathcal{L} \cup \left\{\hat{\mathbf{d}}^T\mathbf{x}_{ij} / \|\hat{\mathbf{d}}\| \geq r_\ell(l_{t,i,j}+v_{t,i})\right\}$
    \ENDFOR
    \STATE \textit{// 步骤 5：判断层内停滞}
    \IF{$|\mathcal{H}| \geq W_{\text{stag}}$}
        \STATE $\mathcal{H}_{\text{recent}} \leftarrow \text{LAST\_N\_ELEMENTS}(\mathcal{H}, W_{\text{stag}})$
        \STATE $\text{improvements} \leftarrow \{\mathcal{H}_{\text{recent}}[i] - \mathcal{H}_{\text{recent}}[i+1] \mid i = 1,\ldots,W_{\text{stag}}-1\}$
        \IF{$\max(\text{improvements}) < \epsilon_{\text{stag}}$}
            \STATE \textit{// 步骤 6：层次提升}
            \IF{$\ell < L_{\max}$}
                \STATE $\ell \leftarrow \ell + 1$,$\ r_\ell \leftarrow r_{\min} + (\ell-1)/(L_{\max}-1) \cdot (1 - r_{\min})$ \COMMENT{层内停滞，提升层次}
                \STATE $\mathcal{D}_\ell \leftarrow \text{GENERATE\_GLOBAL\_DIRECTIONS}(\ell)$
                \FORALL{$(i,j) \in E,\ t \in [0,T-1],\ \mathbf{d} \in \mathcal{D}_\ell$}
                    \STATE $\mathcal{L} \leftarrow \mathcal{L} \cup \left\{\mathbf{d}^T\mathbf{x}_{ij} / \|\mathbf{d}\| \geq r_\ell(l_{t,i,j}+v_{t,i})\right\}$
                \ENDFOR
                \STATE $\mathcal{H} \leftarrow \emptyset$ \COMMENT{清空历史，开始新层监测}
            \ELSE
                \STATE \textbf{break}
            \ENDIF
        \ENDIF
    \ENDIF
\ENDWHILE
\STATE \textit{// 阶段三：终止处理}
\RETURN $(P^k, Q^k, l^k, v^k)$，\textsc{NotFullyConverged}
\end{algorithmic}
\end{breakablealgorithm}

\subsection{算法复杂度分析}

\subsubsection{时间复杂度}

\textbf{决策变量规模。}
问题的决策变量共有
\begin{equation}
n = \underbrace{3T|E|}_{\text{支路变量}\ P, Q, l}
  + \underbrace{3T|N|}_{\text{节点变量}\ v, p, q}
  + \underbrace{2T|G|}_{\text{发电机变量}}
  + \underbrace{T|DR|}_{\text{可调负荷变量}}
= O\!\left(T \cdot |N|\right)
\end{equation}
个，其中对辐射状（树形）配电网有 $|E| = |N|-1$，且通常 $|G|, |DR| \ll |N|$，故 $n = O(T|N|) = O(T|E|)$。

\textbf{单次迭代各步骤代价。}
初始约束数（原始 SOCP，不含割集合 $\mathcal{L}$）为
\[
m_0 = O\!\left(T \cdot (|E|+|N|)\right) = O(n),
\]
包括功率平衡等式 ($2T|N|$)、电压方程等式 ($T|E|$)、二阶锥约束 ($T|E|$) 及各类上下界约束。
内点法求解 SOCP 时，每次牛顿步需对规模 $n \times n$ 的法方程组进行 Cholesky 分解，代价为 $O(n^2 m + n^3)$；当 $m = O(n)$ 时单步代价为 $O(n^3)$，内点法通常在 $O(1)$--$O(\sqrt{n})$ 步内收敛，整体近似为 $O(n^3)$。
间隙计算（步骤 2）遍历所有 $T|E|$ 个支路--时段对，代价 $O(T|E|)$；
方向割生成（步骤 4）同样遍历一次，代价 $O(T|E|)$。
两者均满足 $T|E| = O(n) \ll n^3$，对总代价贡献可忽略。

\textbf{累积割对后续迭代的影响。}
第 $k$ 次迭代时，约束集合 $\mathcal{L}$ 已累积至多 $k \cdot T|E|$ 条割，约束总数为
\[
m_k = O\!\left(n + k \cdot T|E|\right) = O(k \cdot n).
\]
对应的 SOCP 求解代价上升至 $O(n^2 m_k + n^3) = O(k \cdot n^3)$。
全部 $K_{\max}$ 次迭代的总时间代价为
\[
\sum_{k=1}^{K_{\max}} O(k \cdot n^3)
= O\!\left(K_{\max}^2 \cdot n^3\right).
\]
由于 $K_{\max}$ 为小常数（默认 50），实践中可视为 $O(K_{\max}^2 \cdot n^3)$。

\subsubsection{空间复杂度}

\textbf{决策变量存储。}
支路变量 $P, Q, l$ 共 $3T|E|$ 个，节点变量 $v, p, q$ 共 $3T|N|$ 个，合计 $O(T \cdot (|E|+|N|)) = O(T|N|) = O(T|E|)$（辐射网）。

\textbf{方向割约束集合 $\mathcal{L}$ 的存储。}
每条割约束形如
\[
\hat{\mathbf{d}}^T \mathbf{x}_{ij} \geq r_\ell\,(l_{t,i,j}+v_{t,i}),
\]
仅涉及某条支路 $(i,j)$ 在时段 $t$ 的 3--4 个变量，以稀疏线性约束存储时每条代价 $O(1)$。
自适应割（步骤 4）：每轮至多新增 $T|E|$ 条，$K_{\max}$ 轮后累积不超过 $K_{\max} \cdot T|E|$ 条。
全局方向集割（步骤 6 层切换）：每次层切换对全部 $T|E|$ 个支路--时段对添加 $|\mathcal{D}_\ell|$ 条割，$L_{\max}$ 次切换共增加 $\sum_{\ell} |\mathcal{D}_\ell| \cdot T|E|$ 条；由于 $L_{\max}$ 及各层方向数 $|\mathcal{D}_\ell|$ 均为固定小常数，此项可并入 $O(K_{\max} \cdot T|E|)$ 量级。

间隙历史集合 $\mathcal{H}$ 每次层切换后清空，始终仅保留 $W_{\text{stag}}$ 个标量，占用 $O(1)$ 空间。

\subsubsection{迭代次数分析}

\textbf{间隙序列的单调性。}
每次向 $\mathcal{L}$ 加入新割后，可行域单调缩小，下一次 SOCP 的最优解受到更严格约束。由于新割的方向正是当前松弛间隙最大的支路方向，割的添加直接限制了该方向上的自由度，实践中可观察到最大间隙序列 $\{\Delta^k_{\max}\}$ 具有单调不增趋势。

\textbf{层次结构提供的终止上界。}
算法最多执行 $L_{\max}$ 次层切换。每层内，若连续 $W_{\text{stag}}$ 次迭代的间隙改善量均低于 $\epsilon_{\text{stag}}$，则触发层切换，因此每层消耗的「停滞迭代」不超过 $W_{\text{stag}}$ 次。层切换引入的全局方向集割覆盖了更大角度范围，显著增大约束覆盖度，从而加速下一层的收敛。层切换终止的总迭代上界为 $L_{\max} \cdot W_{\text{stag}}$（纯停滞情形下），结合各层内的有效收敛迭代，总迭代次数约为
\[
K_{\text{est}} \lesssim L_{\max} \cdot (W_{\text{stag}} + K_{\text{converge per layer}}),
\]
其中 $K_{\text{converge per layer}}$ 为单层内达到收敛或触发停滞所需的有效迭代数。

\textbf{典型参数下的估算。}
取 $L_{\max} = 3$，$W_{\text{stag}} = 5$，$K_{\text{converge per layer}} \approx 5$--$10$，则估算总迭代数为 $3 \times (5 + 7.5) \approx 37.5$，与经验值 10--30 次吻合（辐射网 SOCP 松弛通常已接近紧，初始间隙较小，实际迭代数偏少）。
最坏情况受 $K_{\max} = 50$ 强制截断，不会超出。

\begin{itemize}
\item \textbf{典型情况}：10--30 次迭代（辐射网松弛间隙小，少量割即可收紧）
\item \textbf{层切换上界}：$L_{\max} \cdot W_{\text{stag}}$（纯停滞路径）
\item \textbf{硬上界}：$K_{\max}$（算法强制终止条件）
\item \textbf{层数}：通常触发 2--3 次层切换，即停留于第 2--3 层时收敛
\end{itemize}
